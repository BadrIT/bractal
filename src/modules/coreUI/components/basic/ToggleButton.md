### Basic Example

```js
  <Examples>
    <ToggleButton>Click This</ToggleButton>
    <ToggleButton selected>Click This</ToggleButton>
    <ToggleButton disabled>Click This</ToggleButton>    
  </Examples>
```

### Type (Primary/Secondary)

```js
  <Examples>
    <VerticalExamples>
      <ToggleButton>Click This</ToggleButton>
      <ToggleButton secondary>Click This</ToggleButton>
    </VerticalExamples>
    <VerticalExamples>
      <ToggleButton selected>Click This</ToggleButton>
      <ToggleButton secondary selected>Click This</ToggleButton>
    </VerticalExamples>
    <VerticalExamples>
      <ToggleButton disabled>Click This</ToggleButton>    
      <ToggleButton secondary disabled>Click This</ToggleButton>    
    </VerticalExamples>
  </Examples>
```
